28 /*
29  * class Trituple is used to store a node's information in SparseMatrix
30  */
31 template <typename Scalar>
32 class Trituple
33 {
34 public:
35  Trituple():row_(0),col_(0),value_(0){}
36  Trituple(unsigned int row, unsigned int col, Scalar value)
37  :row_(row),col_(col),value_(value){}
38  bool operator==(const Trituple<Scalar> &tri2) const
39  {
40  return (tri2.row_ == row_ && tri2.col_ == col_ && tri2.value_ == value_);
41  }
42  bool operator!=(const Trituple<Scalar> &tri2) const
43  {
44  return (tri2.row_ != row_ || tri2.col_ != col_ || tri2.value_ != value_);
45  }
46  inline unsigned int row() const { return row_;}
47  inline unsigned int col() const { return col_;}
48  inline Scalar value() const { return value_;}
49  inline void setRow(unsigned int i){row_ = i;}
50  inline void setCol(unsigned int j){col_ = j;}
51  inline void setValue(Scalar k){value_ = k;}
52 private:
53  unsigned int row_;
54  unsigned int col_;
55  Scalar value_;
56 };
