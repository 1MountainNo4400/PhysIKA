28 #ifndef NDEBUG
29 # define PHYSIKA_MESSAGE_ASSERT(condition,message) \
30  do \
31  {\
32  if(!(condition)) std::cerr<<message<<std::endl; \
33  assert((condition)); \
34  }while(false)

36 # define PHYSIKA_MESSAGE_ASSERT(condition,message) do{}while(false)

40 #ifndef NDEBUG
41 # define PHYSIKA_ERROR(message) \
42  do \
43  { \
44  std::cerr<<message<<std::endl; \
45  std::cerr<<"Assertion failed: function "<<__FUNCTION__<<", file "<<__FILE__<<", line "<<__LINE__<<"."<<std::endl; \
46  std::exit(EXIT_FAILURE); \
47  }while(false)
48 #else
49 # define PHYSIKA_ERROR(message) do{}while(false)
