23 //is_same: if two types are the same, value is true; otherwise, false
24 //the name 'is_same' violates Physika naming rule in order to agree with std::is_same (C++11)
25 template <typename T1, typename T2> struct is_same { static const bool value = false; };
26 template <typename T> struct is_same<T,T> { static const bool value = true; };
