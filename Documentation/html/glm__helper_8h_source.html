22 inline __host__ __device__ void JacobiRotate(glm::mat3 &A, glm::mat3 &R, int p, int q)
23 {
24  // rotates A through phi in pq-plane to set A(p,q) = 0
25  // rotation stored in R whose columns are eigenvectors of A
26  if (A[p][q] == 0.0f)
27  return;
28 
29  float d = (A[p][p] - A[q][q]) / (2.0f*A[p][q]);
30  float t = 1.0f / (fabs(d) + sqrt(d*d + 1.0f));
31  if (d < 0.0f) t = -t;
32  float c = 1.0f / sqrt(t*t + 1);
33  float s = t*c;
34  A[p][p] += t*A[p][q];
35  A[q][q] -= t*A[p][q];
36  A[p][q] = 0.0f;
37  A[q][p] = 0.0f;
38  // transform A
39  int k;
40  for (k = 0; k < 3; k++) {
41  if (k != p && k != q) {
42  float Akp = c*A[k][p] + s*A[k][q];
43  float Akq = -s*A[k][p] + c*A[k][q];
44  A[k][p] = Akp;
45  A[p][k] = Akp;
46  A[k][q] = Akq;
47  A[q][k] = Akq;
48  }
49  }
50  // store rotation in R
51  for (k = 0; k < 3; k++) {
52  float Rkp = c*R[k][p] + s*R[k][q];
53  float Rkq = -s*R[k][p] + c*R[k][q];
54  R[k][p] = Rkp;
55  R[k][q] = Rkq;
56  }
57 }

59 inline __host__ __device__ void EigenDecomposition(const glm::mat3 &A, glm::mat3 &eigenVecs, glm::vec3 &eigenVals)
60 {
61  const int numJacobiIterations = 10;
62  const float epsilon = 1e-15f;
63 
64  glm::mat3 D = A;
65 
66  // only for symmetric matrices!
67  eigenVecs = glm::mat3(); // unit matrix
68  int iter = 0;
69  while (iter < numJacobiIterations) { // 3 off diagonal elements
70  // find off diagonal element with maximum modulus
71  int p, q;
72  float a, max;
73  max = fabs(D[0][1]);
74  p = 0; q = 1;
75  a = fabs(D[0][2]);
76  if (a > max) { p = 0; q = 2; max = a; }
77  a = fabs(D[1][2]);
78  if (a > max) { p = 1; q = 2; max = a; }
79  // all small enough -> done
80  if (max < epsilon) break;
81  // rotate matrix with respect to that element
82  JacobiRotate(D, eigenVecs, p, q);
83  iter++;
84  }
85  eigenVals[0] = D[0][0];
86  eigenVals[1] = D[1][1];
87  eigenVals[2] = D[2][2];
88 }

101  AAT[2][2] = A[2][0] * A[2][0] + A[2][1] * A[2][1] + A[2][2] * A[2][2];
102 
103  AAT[0][1] = A[0][0] * A[1][0] + A[0][1] * A[1][1] + A[0][2] * A[1][2];
104  AAT[0][2] = A[0][0] * A[2][0] + A[0][1] * A[2][1] + A[0][2] * A[2][2];
105  AAT[1][2] = A[1][0] * A[2][0] + A[1][1] * A[2][1] + A[1][2] * A[2][2];
106 
107  AAT[1][0] = AAT[0][1];
108  AAT[2][0] = AAT[0][2];
109  AAT[2][1] = AAT[1][2];

112  glm::vec3 eigenVals;
113  EigenDecomposition(AAT, U, eigenVals);
114 
115  float d0 = sqrt(eigenVals[0]);
116  float d1 = sqrt(eigenVals[1]);
117  float d2 = sqrt(eigenVals[2]);
118  D = glm::mat3(0.0f);
119  D[0][0] = d0;
120  D[1][1] = d1;
121  D[2][2] = d2;
122

124

126  float l1 = eigenVals[1]; if (l1 <= eps) l1 = 0.0f; else l1 = 1.0f / d1;
127  float l2 = eigenVals[2]; if (l2 <= eps) l2 = 0.0f; else l2 = 1.0f / d2;
128 
129  glm::mat3 S1;
130  S1[0][0] = l0*U[0][0] * U[0][0] + l1*U[0][1] * U[0][1] + l2*U[0][2] * U[0][2];
131  S1[1][1] = l0*U[1][0] * U[1][0] + l1*U[1][1] * U[1][1] + l2*U[1][2] * U[1][2];
132  S1[2][2] = l0*U[2][0] * U[2][0] + l1*U[2][1] * U[2][1] + l2*U[2][2] * U[2][2];
133 
134  S1[0][1] = l0*U[0][0] * U[1][0] + l1*U[0][1] * U[1][1] + l2*U[0][2] * U[1][2];
135  S1[0][2] = l0*U[0][0] * U[2][0] + l1*U[0][1] * U[2][1] + l2*U[0][2] * U[2][2];
136  S1[1][2] = l0*U[1][0] * U[2][0] + l1*U[1][1] * U[2][1] + l2*U[1][2] * U[2][2];
137 
138  S1[1][0] = S1[0][1];
139  S1[2][0] = S1[0][2];
140  S1[2][1] = S1[1][2];
141 
142  R = A*S1;
143 
144  // stabilize
145  glm::vec3 c0, c1, c2;
146  c0 = R[0];
147  c1 = R[1];
148  c2 = R[2];
149 
150  if (c0.length() < eps)
151  c0 = glm::cross(c1, c2);
152  else if (c1.length() < eps)
153  c1 = glm::cross(c2, c0);
154  else
155  c2 = glm::cross(c0, c1);
156  R[0] = c0;
157  R[1] = c1;
158  R[2] = c2;
159 }

189 inline __host__ __device__ void PolarDecompositionStable(const glm::mat3 &M, const float tolerance, glm::mat3 &R)
190 {
191  glm::mat3 Mt = glm::transpose(M);
192  float Mone = OneNorm(M);
193  float Minf = InfNorm(M);
194  float Eone;
195  glm::mat3 MadjTt, Et;
196  do
197  {
198  MadjTt[0] = glm::cross(Mt[1], Mt[2]);
199  MadjTt[1] = glm::cross(Mt[2], Mt[0]);
200  MadjTt[2] = glm::cross(Mt[0], Mt[1]);
201 
202  float det = Mt[0][0] * MadjTt[0][0] + Mt[0][1] * MadjTt[0][1] + Mt[0][2] * MadjTt[0][2];
203 
204  if (fabs(det) < 1.0e-12)
205  {
206  glm::vec3 len;
207  unsigned int index = 0xffffffff;
208  for (unsigned int i = 0; i < 3; i++)
209  {
210  len[i] = MadjTt[i].length();
211  if (len[i] > 1.0e-12)
212  {
213  // index of valid cross product
214  // => is also the index of the vector in Mt that must be exchanged
215  index = i;
216  break;
217  }
218  }
219  if (index == 0xffffffff)
220  {
221  R = glm::mat3();
222  return;
223  }
224  else
225  {
226  Mt[index] = glm::cross(Mt[(index + 1) % 3], Mt[(index + 2) % 3]);
227  MadjTt[(index + 1) % 3] = glm::cross(Mt[(index + 2) % 3], Mt[(index) % 3]);;
228  MadjTt[(index + 2) % 3] = glm::cross(Mt[(index) % 3], Mt[(index + 1) % 3]);
229  glm::mat3 M2 = glm::transpose(Mt);
230  Mone = OneNorm(M2);
231  Minf = InfNorm(M2);
232  det = Mt[0][0] * MadjTt[0][0] + Mt[0][1] * MadjTt[0][1] + Mt[0][2] * MadjTt[0][2];
233  }
234  }
235 
236  const float MadjTone = OneNorm(MadjTt);
237  const float MadjTinf = InfNorm(MadjTt);
238 
239  const float gamma = sqrt(sqrt((MadjTone*MadjTinf) / (Mone*Minf)) / fabs(det));
240 
241  const float g1 = gamma*0.5f;
242  const float g2 = 0.5f / (gamma*det);
243 
244  for (unsigned char i = 0; i < 3; i++)
245  {
246  for (unsigned char j = 0; j < 3; j++)
247  {
248  Et[i][j] = Mt[i][j];
249  Mt[i][j] = g1*Mt[i][j] + g2*MadjTt[i][j];
250  Et[i][j] -= Mt[i][j];
251  }
252  }
253 
254  Eone = OneNorm(Et);
255 
256  Mone = OneNorm(Mt);
257  Minf = InfNorm(Mt);
258  } while (Eone > Mone * tolerance);
259 
260  // Q = Mt^T
261  R = glm::transpose(Mt);
262 }
