24 /*
25  * LinearWeightFunction: the most common linear weight function
26  * f(x,R) = a(1 - |x|/R) (0<=|x|<=R), where the value of 'a' depends on
27  * the dimension and radius of support domain such that partition
28  * of unity is satisfied:
29  * a = 1/R, in 1D
30  * a = 3/(PI*R^2), in 2D
31  * a = 3/(PI*R^3), in 3D
32  */
33 
34 template <typename Scalar, int Dim>
35 class LinearWeightFunction: public WeightFunction<Scalar,Dim>
36 {
37 public:
38  LinearWeightFunction(){}
39  ~LinearWeightFunction(){}
40  Scalar weight(const Vector<Scalar,Dim> &center_to_x, Scalar R) const;
41  Vector<Scalar,Dim> gradient(const Vector<Scalar,Dim> &center_to_x, Scalar R) const;
42  Scalar laplacian(const Vector<Scalar,Dim> &center_to_x, Scalar R) const;
43  void printInfo() const;
44 };
