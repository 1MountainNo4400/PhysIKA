59 //use element-wise dictionary comparison
60 //vectors must be equal in size
61 template <typename Scalar>
62 class CompareVector
63 {
64 public:
65  //return true if v1 < v2
66  bool operator()(const std::vector<Scalar> &v1, const std::vector<Scalar> &v2) const
67  {
68  unsigned int size1 = v1.size();
69  unsigned int size2 = v2.size();
70  if(size1 != size2)
71  throw PhysikaException("CompareVector cannot compare vectors of different sizes!");
72  for(unsigned int i = 0; i < size1; ++i)
73  {
74  if(v1[i] < v2[i])
75  return true;
76  if(v1[i] > v2[i])
77  return false;
78  }
79  return false;
80  }
81 };
